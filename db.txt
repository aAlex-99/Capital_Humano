CREATE TABLE PROSPECTO(
ID NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
NOMBRE NVARCHAR2(255),
APELLIDOS NVARCHAR2(255),
EMAIL NVARCHAR2(255),
FECHA_NAC DATE,
RFC NVARCHAR2(255),
TELEFONO NVARCHAR2(255),
DOM_ID NUMBER
);

CREATE TABLE EMPLEO(
EMP_ID NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
EMPRESA NVARCHAR2(255),
FECHA_INGRESO DATE,
FECHA_SALIDA DATE,
INGRESO_MENSUAL NUMBER(7,2),
GIRO_EMPRESA NVARCHAR2(255)
);


CREATE TABLE DOMICILIO(
DOM_ID NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
CALLE NVARCHAR2(255),
NUMERO NVARCHAR2(255),
COLONIA NVARCHAR2(255),
CIUDAD NVARCHAR2(255),
ESTADO NVARCHAR2(255),
PAIS NVARCHAR2(255),
CODIGO_POSTAL NVARCHAR2(255)
);

CREATE TABLE PROS_EMPLEO(
REL_ID NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
PROSPECTO_ID NUMBER,
EMPLEDO_ID NUMBER
);

ALTER TABLE PROSPECTO ADD CONSTRAINT FK_PROSPECTO_DOM_ID
FOREIGN KEY(DOM_ID)
REFERENCES DOMICILIO(DOM_ID) ON DELETE CASCADE;

ALTER TABLE PROS_EMPLEO ADD CONSTRAINT FK_PROS_EMP_PROS_ID
FOREIGN KEY(PROSPECTO_ID)
REFERENCES PROSPECTO(ID) ON DELETE CASCADE;

ALTER TABLE PROS_EMPLEO ADD CONSTRAINT FK_PROS_EMP_EMPLEO_ID
FOREIGN KEY(EMPLEDO_ID)
REFERENCES EMPLEO(EMP_ID);
